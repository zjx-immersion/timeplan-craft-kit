### 3.2 快捷键优先级建议

#### 第一优先级（P0 - 必须有）

```typescript
// 推荐实现方案
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    // Ctrl+Z 撤销
    if (e.ctrlKey && e.key === 'z' && !e.shiftKey) {
      e.preventDefault();
      undo();
    }
    
    // Ctrl+Y 或 Ctrl+Shift+Z 重做
    if ((e.ctrlKey && e.key === 'y') || 
        (e.ctrlKey && e.shiftKey && e.key === 'z')) {
      e.preventDefault();
      redo();
    }
    
    // Ctrl+S 保存
    if (e.ctrlKey && e.key === 's') {
      e.preventDefault();
      handleSave();
    }
  };
  
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [undo, redo, handleSave]);
```

#### 第二优先级（P1 - 应该有）

```typescript
// 视图切换快捷键
if (e.ctrlKey && e.key >= '1' && e.key <= '5') {
```

```typescript
useEffect(() => {
  const handleGlobalKeyDown = (e: KeyboardEvent) => {
    // 忽略输入框中的按键
    if ((e.target as HTMLElement).tagName.match(/INPUT|TEXTAREA/)) {
      return;
    }
    
    // Ctrl+Z 撤销
    if (e.ctrlKey && e.key === 'z' && !e.shiftKey) {
      e.preventDefault();
      undo();
      message.info('已撤销');
    }
    
    // Ctrl+Y 重做
    if (e.ctrlKey && e.key === 'y') {
      e.preventDefault();
      redo();
      message.info('已重做');
    }
    
    // Ctrl+S 保存
    if (e.ctrlKey && e.key === 's') {
      e.preventDefault();
      handleSave();
    }
    
    // Space 定位今日
    if (e.key === ' ') {
      e.preventDefault();
      scrollToToday();
    }
    
    // Ctrl+数字 切换视图
    if (e.ctrlKey && e.key >= '1' && e.key <= '5') {
      e.preventDefault();
      const scales = ['day', 'week', 'month', 'quarter', 'biweekly'];
      handleScaleChange(scales[parseInt(e.key) - 1]);
    }
  };
  
  window.addEventListener('keydown', handleGlobalKeyDown);
  return () => window.removeEventListener('keydown', handleGlobalKeyDown);
}, [undo, redo, handleSave, scrollToToday, handleScaleChange]);
```
